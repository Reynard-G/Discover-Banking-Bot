CREATE TABLE `accounts` (
	`id` BIGINT(20) unsigned NOT NULL AUTO_INCREMENT,
	`user_id` BIGINT(20) NOT NULL,
	`username` VARCHAR(191) CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci NOT NULL,
	`created_at` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
	`updated_at` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
	PRIMARY KEY (`id`)
) ENGINE=MyISAM;

CREATE TABLE `transactions` (
	`id` BIGINT(20) unsigned NOT NULL AUTO_INCREMENT,
	`user_id` BIGINT(20) NOT NULL,
	`amount` DECIMAL(10,2) NOT NULL,
	`fee` DECIMAL(10,2) NOT NULL DEFAULT 0.00,
	`cr_dr` VARCHAR(2) CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci NOT NULL,
	`status` TINYINT(4) NOT NULL,
	`note` TEXT CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci DEFAULT NULL,
	`creditcard_id` BIGINT(20) DEFAULT NULL,
	`attachment` VARCHAR(191) CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci DEFAULT NULL,
	`created_user_id` BIGINT(20) DEFAULT NULL,
	`updated_user_id` BIGINT(20) DEFAULT NULL,
	`created_at` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
	`updated_at` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
	PRIMARY KEY (`id`)
) ENGINE=MyISAM;
